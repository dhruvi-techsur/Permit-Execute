```typescript
  apiClient.post<Document>(`${base(permitId)}/confirm`, { documentId }).then(r => r.data);

export const listDocuments = (permitId: string) =>
  apiClient.get<Document[]>(base(permitId)).then(r => r.data);

export const getDownloadUrl = (permitId: string, docId: string) =>
  apiClient.get<DownloadResponse>(`${base(permitId)}/${docId}/download`).then(r => r.data);

export const deleteDocument = (permitId: string, docId: string) =>
  apiClient.delete(`${base(permitId)}/${docId}`);

// Convenience: upload a File object directly using a presigned URL (bypasses apiClient — raw fetch to S3)
export const uploadToS3 = (uploadUrl: string, file: File, onProgress?: (pct: number) => void): Promise<void> =>
  new Promise((resolve, reject) => {
    const xhr = new XMLHttpRequest();
    xhr.open('PUT', uploadUrl);
    xhr.setRequestHeader('Content-Type', file.type);
    if (onProgress) {
      xhr.upload.onprogress = e => { if (e.lengthComputable) onProgress(Math.round((e.loaded / e.total) * 100)); };
    }
    xhr.onload = () => (xhr.status >= 200 && xhr.status < 300 ? resolve() : reject(new Error(`S3 upload failed: ${xhr.status}`)));
    xhr.onerror = () => reject(new Error('S3 upload network error'));
    xhr.send(file);
  });
```

```typescript
import { create } from 'zustand';
import type { PermitApplication, PermitListQuery, PaginatedResponse } from '../types/permit.types';
import { listPermits, getPermit } from '../api/permits.api';

interface PermitsState {
  permits: PermitApplication[];
  selectedPermit: PermitApplication | null;
  pagination: { total: number; page: number; limit: number } | null;
  isLoading: boolean;
  error: string | null;
  fetchPermits: (query?: PermitListQuery) => Promise<void>;
  setSelectedPermit: (permit: PermitApplication | null) => void;
  fetchPermit: (id: string) => Promise<PermitApplication>;
  updatePermitInList: (permit: PermitApplication) => void;
}

export const usePermitsStore = create<PermitsState>((set, get) => ({
  permits: [],
  selectedPermit: null,
  pagination: null,
  isLoading: false,
  error: null,

  fetchPermits: async (query = {}) => {
    set({ isLoading: true, error: null });
    try {
      const res = await listPermits(query);
      set({ permits: res.data, pagination: res.meta, isLoading: false });
    } catch (err: unknown) {
      set({ isLoading: false, error: (err as Error).message });
    }
  },

  setSelectedPermit: (permit) => set({ selectedPermit: permit }),

  fetchPermit: async (id) => {
    const permit = await getPermit(id);
    set({ selectedPermit: permit });
    return permit;
  },

  updatePermitInList: (permit) =>
    set(state => ({
      permits: state.permits.map(p => p.id === permit.id ? permit : p),
      selectedPermit: state.selectedPermit?.id === permit.id ? permit : state.selectedPermit,
    })),
}));
```

```typescript
import { create } from 'zustand';
import type { Notification } from '../types/notification.types';
import { listNotifications, getUnreadCount, markNotificationRead, markAllNotificationsRead } from '../api/notifications.api';

interface NotificationsState {
  notifications: Notification[];
  unreadCount: number;
  isLoading: boolean;
  fetchUnreadCount: () => Promise<void>;
  fetchNotifications: (page?: number) => Promise<void>;
  markRead: (id: string) => Promise<void>;
  markAllRead: () => Promise<void>;
}

export const useNotificationsStore = create<NotificationsState>((set) => ({
  notifications: [],
  unreadCount: 0,
  isLoading: false,

  fetchUnreadCount: async () => {
    try {
      const { count } = await getUnreadCount();
      set({ unreadCount: count });
    } catch { /* silent fail for polling */ }
  },

  fetchNotifications: async (page = 1) => {
    set({ isLoading: true });
    try {
      const res = await listNotifications(page);
      set({ notifications: res.data, isLoading: false });
    } catch { set({ isLoading: false }); }
  },

  markRead: async (id) => {
    await markNotificationRead(id);
    set(state => ({
      notifications: state.notifications.map(n => n.id === id ? { ...n, isRead: true } : n),
      unreadCount: Math.max(0, state.unreadCount - 1),
    }));
  },

  markAllRead: async () => {
    await markAllNotificationsRead();
    set(state => ({
      notifications: state.notifications.map(n => ({ ...n, isRead: true })),
      unreadCount: 0,
    }));
  },
}));
```

```typescript
import { useEffect, useCallback, useState } from 'react';
import { getPermit } from '../api/permits.api';
import { usePermitsStore } from '../store/permits.store';
import type { PermitApplication } from '../types/permit.types';

export function usePermit(id: string) {
  const { selectedPermit, fetchPermit } = usePermitsStore();
  const [isLoading, setIsLoading] = useState(false);
  const [error, setError] = useState<string | null>(null);

  const refresh = useCallback(async () => {
    try {
      await fetchPermit(id);
    } catch (err: unknown) {
      setError((err as Error).message);
    }
  }, [id, fetchPermit]);

  // Initial load
  useEffect(() => {
    setIsLoading(true);
    setError(null);
    refresh().finally(() => setIsLoading(false));
  }, [id]);

  // 15-second polling
  useEffect(() => {
    const interval = setInterval(() => { refresh(); }, 15_000);
    return () => clearInterval(interval);
  }, [refresh]);

  return { permit: selectedPermit, isLoading, error, refresh };
}
```

**`frontend/src/hooks/useDocumentUpload.ts`** — orchestrates presign → S3 PUT → confirm:
```typescript
import { useState, useCallback } from 'react';
import type { UploadState } from '../types/document.types';
import { presignDocument, uploadToS3, confirmUpload } from '../api/documents.api';

const MAX_FILE_SIZE = 26_214_400; // 25MB
const MAX_TOTAL_SIZE = 104_857_600; // 100MB
const MAX_DOCS = 20;
const ALLOWED_TYPES = ['application/pdf', 'image/jpeg', 'image/png', 'application/vnd.openxmlformats-officedocument.wordprocessingml.document'];

export function useDocumentUpload(permitId: string, existingCount = 0) {
  const [uploads, setUploads] = useState<UploadState[]>([]);

  const updateUpload = (file: File, patch: Partial<UploadState>) =>
    setUploads(prev => prev.map(u => u.file === file ? { ...u, ...patch } : u));

  const uploadFile = async (file: File) => {
    // Client-side validation
    if (!ALLOWED_TYPES.includes(file.type)) {
      updateUpload(file, { status: 'error', error: `File type not allowed: ${file.type}` });
      return;
    }
    if (file.size > MAX_FILE_SIZE) {
      updateUpload(file, { status: 'error', error: `File exceeds 25MB limit (${(file.size / 1024 / 1024).toFixed(1)}MB)` });
      return;
    }

    try {
      updateUpload(file, { status: 'uploading', progress: 0 });

      // Step 1: Presign
      const { documentId, uploadUrl } = await presignDocument(permitId, {
        fileName: file.name,
        mimeType: file.type,
        fileSizeBytes: file.size,
      });

      // Step 2: PUT to S3/MinIO
      await uploadToS3(uploadUrl, file, (pct) => updateUpload(file, { progress: pct }));

      // Step 3: Confirm
      await confirmUpload(permitId, documentId);

      updateUpload(file, { status: 'uploaded', progress: 100, documentId });
    } catch (err: unknown) {
      updateUpload(file, { status: 'error', error: (err as Error).message });
      // Individual failure does NOT block other files
    }
  };

  const uploadFiles = useCallback(async (files: File[]) => {
    // Validate total size and count
    const totalSize = files.reduce((s, f) => s + f.size, 0);
    if (existingCount + files.length > MAX_DOCS) {
      alert(`Cannot upload: would exceed maximum of ${MAX_DOCS} documents`);
      return;
    }
    if (totalSize > MAX_TOTAL_SIZE) {
      alert('Total upload size exceeds 100MB limit');
      return;
    }

    // Initialize all as queued
    setUploads(files.map(file => ({ file, status: 'queued', progress: 0 })));

    // Upload individually — failures don't block others
    await Promise.allSettled(files.map(uploadFile));
  }, [permitId, existingCount]);

  const clearUploads = useCallback(() => setUploads([]), []);

  const isUploading = uploads.some(u => u.status === 'uploading' || u.status === 'queued');

  return { uploadFiles, uploads, isUploading, clearUploads };
}
```

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-02-14 | Tampering | useDocumentUpload client-side MIME check | accept | Client-side MIME validation in `useDocumentUpload` is a UX guard only — server-side `PresignRequestDto` is the authoritative enforcement. Client check reduces unnecessary API calls; residual bypass risk accepted (server enforces). |
| T-02-15 | Information disclosure | Zustand store — accessToken | accept | `usePermitsStore` and `useNotificationsStore` do not store auth tokens — auth store is separate in Phase 1 design. apiClient interceptor reads token from auth store. No token in permits or notifications state. |
| T-02-16 | Denial of service | 10s/15s polling | accept | Polling intervals are reasonable for v1 (10s notifications, 15s permit detail). No exponential backoff for polling errors in this phase — accepted for v1; v2 can add SSE. Silent fail on polling errors prevents UX disruption. |
</threat_model>
