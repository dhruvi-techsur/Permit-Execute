**`PermitStatusBadge.tsx`** — Colored pill badge for each of the 6 status values using Tailwind:
```typescript
import type { PermitStatus } from '../../types/permit.types';

const STATUS_CONFIG: Record<PermitStatus, { label: string; className: string }> = {
  draft:                  { label: 'Draft',               className: 'bg-gray-100 text-gray-700 ring-1 ring-gray-200' },
  submitted:              { label: 'Submitted',           className: 'bg-blue-50 text-blue-700 ring-1 ring-blue-200' },
  under_review:           { label: 'Under Review',        className: 'bg-amber-50 text-amber-700 ring-1 ring-amber-200' },
  additional_info_needed: { label: 'Info Needed',         className: 'bg-orange-50 text-orange-700 ring-1 ring-orange-200' },
  approved:               { label: 'Approved',            className: 'bg-green-50 text-green-700 ring-1 ring-green-200' },
  rejected:               { label: 'Rejected',            className: 'bg-red-50 text-red-700 ring-1 ring-red-200' },
};

export function PermitStatusBadge({ status, className = '' }: { status: PermitStatus; className?: string }) {
  const { label, className: statusClass } = STATUS_CONFIG[status];
  return (
    <span className={`inline-flex items-center rounded-full px-2.5 py-0.5 text-xs font-medium ${statusClass} ${className}`}>
      {label}
    </span>
  );
}
```

import { test, expect } from '@playwright/test';

// Helper: login as applicant
async function loginAsApplicant(page) {
  await page.goto('/login');
  await page.getByLabel('Email').fill('applicant@permitflow.test');
  await page.getByLabel('Password').fill('password');

test.describe('PermitListPage', () => {
  test.beforeEach(async ({ page }) => {
    await loginAsApplicant(page);
  });

  test('navigates to /applications from sidebar', async ({ page }) => {
    await page.getByRole('link', { name: /my applications/i }).click();
    await expect(page).toHaveURL('/applications');
    await expect(page.getByRole('heading', { name: /my applications/i })).toBeVisible();
  });

  test('shows skeleton loaders while fetching', async ({ page }) => {
    // Intercept the API to delay response
    await page.route('**/api/v1/permits*', async route => {
      await new Promise(r => setTimeout(r, 500));
      await route.continue();
    });
    await page.goto('/applications');
    // Skeleton (animate-pulse) should be visible before data loads
    await expect(page.locator('[data-testid="permit-card-skeleton"]').first()).toBeVisible();

  test('shows empty state when no applications exist', async ({ page }) => {
    await page.route('**/api/v1/permits*', route =>
      route.fulfill({ json: { data: [], meta: { total: 0, page: 1, limit: 20 } } })

    await page.goto('/applications');
    await expect(page.getByText(/start your first application/i)).toBeVisible();
