```typescript
@Injectable()
export class DashboardService {
  constructor(private readonly dataSource: DataSource) {}

  async getApplicantDashboard(userId: string): Promise<ApplicantDashboardDto> {
    const qr = this.dataSource.createQueryRunner();
    try {
      // 1. Permit counts by status (applicant's permits only)
      const statusCounts: { status: string; count: string }[] = await qr.query(`
        SELECT status, COUNT(*)::int AS count
        FROM permit_applications
        WHERE applicant_id = $1
        GROUP BY status
        ORDER BY status
      `, [userId]);

      // 2. Active permit count (not draft, approved, or rejected)
      const activeStatuses = ['submitted', 'under_review', 'additional_info_needed'];
      const activePermitCount = statusCounts
        .filter(r => activeStatuses.includes(r.status))
        .reduce((sum, r) => sum + Number(r.count), 0);

      // 3. Recent status changes (last 5 across all applicant's permits)
      const recentChanges: { application_id: string; title: string; to_status: string; created_at: string }[] = await qr.query(`
        SELECT psh.application_id, pa.title, psh.to_status, psh.created_at
        FROM permit_status_history psh
        JOIN permit_applications pa ON pa.id = psh.application_id
        WHERE pa.applicant_id = $1
        ORDER BY psh.created_at DESC
        LIMIT 5
      `, [userId]);

      // 4. Total unread messages across all applicant's applications
      //    (messages where sender_id != userId on their applications)
      const unreadResult: { count: string }[] = await qr.query(`
        SELECT COUNT(*)::int AS count
        FROM messages m
        JOIN permit_applications pa ON pa.id = m.application_id
        WHERE pa.applicant_id = $1
          AND m.sender_id != $1
      `, [userId]);

      return {
        activePermitCount,
        permitsByStatus: statusCounts.map(r => ({ status: r.status, count: Number(r.count) })),
        recentStatusChanges: recentChanges.map(r => ({
          applicationId: r.application_id,
          title: r.title,
          toStatus: r.to_status,
          changedAt: r.created_at,
        })),
        totalUnreadMessages: Number(unreadResult[0]?.count ?? 0),
      };
    } finally {
      await qr.release();
    }
  }

  async getReviewerDashboard(userId: string): Promise<ReviewerDashboardDto> {
    const qr = this.dataSource.createQueryRunner();
    try {
      // 1. Count permits assigned to this reviewer
      const [{ assigned_count }]: { assigned_count: string }[] = await qr.query(`
        SELECT COUNT(*)::int AS assigned_count
        FROM permit_applications
        WHERE reviewer_id = $1
      `, [userId]);

      // 2. Pending action count (submitted + under_review assigned to this reviewer)
      const [{ pending_count }]: { pending_count: string }[] = await qr.query(`
        SELECT COUNT(*)::int AS pending_count
        FROM permit_applications
        WHERE reviewer_id = $1
          AND status IN ('submitted', 'under_review', 'additional_info_needed')
      `, [userId]);

      // 3. Status distribution of all non-draft permits (reviewer sees all)
      const statusCounts: { status: string; count: string }[] = await qr.query(`
        SELECT status, COUNT(*)::int AS count
        FROM permit_applications
        WHERE status != 'draft'
        GROUP BY status
        ORDER BY status
      `);

      // 4. 5 most recently updated permits assigned to this reviewer
      const recentApps: { id: string; title: string; status: string; email: string; submission_date: string | null }[] = await qr.query(`
        SELECT pa.id, pa.title, pa.status, u.email, pa.submission_date
        FROM permit_applications pa
        JOIN users u ON u.id = pa.applicant_id
        WHERE pa.reviewer_id = $1
        ORDER BY pa.updated_at DESC
        LIMIT 5
      `, [userId]);

      // 5. Unread messages across all assigned applications
      const [{ unread_count }]: { unread_count: string }[] = await qr.query(`
        SELECT COUNT(*)::int AS unread_count
        FROM messages m
        JOIN permit_applications pa ON pa.id = m.application_id
        WHERE pa.reviewer_id = $1
          AND m.sender_id != $1
      `, [userId]);

      return {
        assignedCount: Number(assigned_count),
        pendingActionCount: Number(pending_count),
        permitsByStatus: statusCounts.map(r => ({ status: r.status, count: Number(r.count) })),
        recentApplications: recentApps.map(r => ({
          id: r.id,
          title: r.title,
          status: r.status,
          applicantEmail: r.email,
          submittedAt: r.submission_date,
        })),
        totalUnreadMessages: Number(unread_count),
      };
    } finally {
      await qr.release();
    }
  }

  async getAdminDashboard(): Promise<AdminDashboardDto> {
    const qr = this.dataSource.createQueryRunner();
    try {
      // 1. Total permit count
      const [{ total }]: { total: string }[] = await qr.query(`SELECT COUNT(*)::int AS total FROM permit_applications`);

      // 2. Status distribution (all permits)
      const statusCounts: { status: string; count: string }[] = await qr.query(`
        SELECT status, COUNT(*)::int AS count
        FROM permit_applications
        GROUP BY status
        ORDER BY status
      `);

      // 3. Reviewer workload
      const reviewerWorkload: { reviewer_id: string; reviewer_name: string; assigned_count: string; pending_count: string }[] = await qr.query(`
        SELECT
          u.id AS reviewer_id,
          u.full_name AS reviewer_name,
          COUNT(pa.id)::int AS assigned_count,
          COUNT(CASE WHEN pa.status IN ('submitted', 'under_review', 'additional_info_needed') THEN 1 END)::int AS pending_count
        FROM users u
        LEFT JOIN permit_applications pa ON pa.reviewer_id = u.id
        WHERE u.role = 'reviewer'
        GROUP BY u.id, u.full_name
        ORDER BY assigned_count DESC
      `);

      // 4. Last 10 status changes system-wide
      const recentActivity: { application_id: string; title: string; to_status: string; created_at: string; full_name: string }[] = await qr.query(`
        SELECT
          psh.application_id,
          pa.title,
          psh.to_status,
          psh.created_at,
          u.full_name
        FROM permit_status_history psh
        JOIN permit_applications pa ON pa.id = psh.application_id
        JOIN users u ON u.id = psh.changed_by
        ORDER BY psh.created_at DESC
        LIMIT 10
      `);

      return {
        totalPermits: Number(total),
        permitsByStatus: statusCounts.map(r => ({ status: r.status, count: Number(r.count) })),
        reviewerWorkload: reviewerWorkload.map(r => ({
          reviewerId: r.reviewer_id,
          reviewerName: r.reviewer_name,
          assignedCount: Number(r.assigned_count),
          pendingCount: Number(r.pending_count),
        })),
        recentActivity: recentActivity.map(r => ({
          applicationId: r.application_id,
          title: r.title,
          toStatus: r.to_status,
          changedAt: r.created_at,
          changedByName: r.full_name,
        })),
      };
    } finally {
      await qr.release();
    }
  }
}
```

@Controller('api/v1/dashboard')
@UseGuards(JwtAuthGuard)
export class DashboardController {
  constructor(private readonly dashboardService: DashboardService) {}

  // GET /api/v1/dashboard/applicant — applicant role only
  @Get('applicant')
  @Roles('applicant')
  @UseGuards(RolesGuard)
  getApplicantDashboard(@Request() req) {
    return this.dashboardService.getApplicantDashboard(req.user.id);
  }

  // GET /api/v1/dashboard/reviewer — reviewer role only
  @Get('reviewer')
  @Roles('reviewer')
  @UseGuards(RolesGuard)
  getReviewerDashboard(@Request() req) {
    return this.dashboardService.getReviewerDashboard(req.user.id);
  }

  // GET /api/v1/dashboard/admin — admin role only
  @Get('admin')
  @Roles('admin')
  @UseGuards(RolesGuard)
  getAdminDashboard() {
    return this.dashboardService.getAdminDashboard();
  }
}

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| client→API | JWT-authenticated requests to dashboard endpoints; userId extracted from verified JWT payload |
| DB→response | Aggregated stats from permit_applications, messages, permit_status_history returned to role-specific callers |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-04-01 | Elevation of privilege | GET /api/v1/dashboard/admin | mitigate | `@Roles('admin') @UseGuards(RolesGuard)` in `dashboard.controller.ts::getAdminDashboard` — applicant/reviewer JWTs receive 403; system-wide stats are not disclosed to non-admins |
| T-04-02 | Elevation of privilege | GET /api/v1/dashboard/reviewer | mitigate | `@Roles('reviewer') @UseGuards(RolesGuard)` in `dashboard.controller.ts::getReviewerDashboard` — applicants receive 403; reviewer sees only their own assigned permit data (userId filtered in SQL WHERE reviewer_id = $1) |
| T-04-03 | Information disclosure | GET /api/v1/dashboard/applicant — unread message count | mitigate | `dashboard.service.ts::getApplicantDashboard` scopes the messages COUNT query with `WHERE pa.applicant_id = $1` — applicant's userId from JWT, never client-supplied; cannot enumerate other applicants' message counts |
| T-04-04 | Tampering | SQL aggregation queries — parameter injection | mitigate | All SQL in `dashboard.service.ts` uses parameterized queries with DataSource.createQueryRunner().query(sql, [param]) — never string interpolation; $1/$2 bind parameters prevent SQL injection |
| T-04-05 | Denial of service | Aggregation queries on large tables | accept | Queries run COUNT GROUP BY on indexed columns (status, reviewer_id, applicant_id); acceptable for v1 scale; residual risk owned by DBA — add materialized views or caching in v2 if needed |
</threat_model>
