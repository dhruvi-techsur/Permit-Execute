```tsx
import type { ReviewerWorkloadRow } from '../../types/dashboard.types';

interface ReviewerWorkloadTableProps {
  rows: ReviewerWorkloadRow[];
}

export function ReviewerWorkloadTable({ rows }: ReviewerWorkloadTableProps) {
  if (rows.length === 0) {
    return (
      <div className="text-center py-8 text-slate-400 text-sm">No reviewer assignments yet</div>
    );
  }

  return (
    <div className="overflow-x-auto">
      <table className="w-full text-sm">
        <thead>
          <tr className="border-b border-slate-200">
            <th className="text-left py-2 px-3 font-medium text-slate-600">Reviewer</th>
            <th className="text-right py-2 px-3 font-medium text-slate-600">Assigned</th>
            <th className="text-right py-2 px-3 font-medium text-slate-600">Pending Action</th>
            <th className="text-right py-2 px-3 font-medium text-slate-600">Load</th>
          </tr>
        </thead>
        <tbody className="divide-y divide-slate-100">
          {rows.map(row => {
            // Load bar: pendingCount / (assignedCount || 1) as a percentage
            const loadPct = row.assignedCount > 0
              ? Math.round((row.pendingCount / row.assignedCount) * 100)
              : 0;
            const loadColor =
              loadPct > 75 ? 'bg-red-500' :
              loadPct > 50 ? 'bg-amber-500' :
              'bg-emerald-500';

            return (
              <tr key={row.reviewerId} className="hover:bg-slate-50 transition-colors">
                <td className="py-3 px-3">
                  <span className="font-medium text-slate-800">{row.reviewerName}</span>
                </td>
                <td className="py-3 px-3 text-right text-slate-700 font-medium">{row.assignedCount}</td>
                <td className="py-3 px-3 text-right text-slate-700">
                  <span className={row.pendingCount > 0 ? 'font-semibold text-amber-700' : 'text-slate-500'}>
                    {row.pendingCount}
                  </span>
                </td>
                <td className="py-3 px-3">
                  {/* Visual load indicator */}
                  <div className="flex items-center gap-2 justify-end">
                    <div className="w-20 h-1.5 rounded-full bg-slate-100 overflow-hidden">
                      <div
                        className={`h-full rounded-full transition-all duration-500 ${loadColor}`}
                        style={{ width: `${loadPct}%` }}
                      />
                    </div>
                    <span className="text-xs text-slate-500 w-8 text-right">{loadPct}%</span>
                  </div>
                </td>
              </tr>
            );
          })}
        </tbody>
      </table>
    </div>
  );
}
```

```tsx
import { useDashboard } from '../hooks/useDashboard';
import { DashboardStatCard } from '../components/dashboard/DashboardStatCard';
import { StatusDistributionChart } from '../components/dashboard/StatusDistributionChart';
import { ReviewerWorkloadTable } from '../components/dashboard/ReviewerWorkloadTable';
import type { AdminDashboardDto } from '../types/dashboard.types';

const STATUS_LABELS: Record<string, string> = {
  draft: 'Draft', submitted: 'Submitted', under_review: 'Under Review',
  additional_info_needed: 'Info Needed', approved: 'Approved', rejected: 'Rejected',
};

function timeAgo(dateStr: string): string {
  const diffMs = Date.now() - new Date(dateStr).getTime();
  const diffMins = Math.floor(diffMs / 60_000);
  if (diffMins < 1) return 'just now';
  if (diffMins < 60) return `${diffMins}m ago`;
  const diffHrs = Math.floor(diffMins / 60);
  if (diffHrs < 24) return `${diffHrs}h ago`;
  return `${Math.floor(diffHrs / 24)}d ago`;
}

export default function AdminDashboardPage() {
  const { data, isLoading, error } = useDashboard();
  const dashboard = data as AdminDashboardDto | null;

  // Skeleton loading
  if (isLoading) {
    return (
      <div className="max-w-6xl mx-auto px-4 sm:px-6 py-8 space-y-6">
        <div className="h-8 bg-slate-100 animate-pulse rounded-lg w-48" />
        <div className="grid grid-cols-1 sm:grid-cols-2 lg:grid-cols-4 gap-4">
          {[1, 2, 3, 4].map(i => <div key={i} className="h-28 bg-slate-100 animate-pulse rounded-xl" />)}
        </div>
        <div className="grid grid-cols-1 lg:grid-cols-2 gap-6">
          <div className="h-48 bg-slate-100 animate-pulse rounded-xl" />
          <div className="h-48 bg-slate-100 animate-pulse rounded-xl" />
        </div>
        <div className="h-64 bg-slate-100 animate-pulse rounded-xl" />
      </div>
    );
  }

  if (error) {
    return (
      <div className="max-w-6xl mx-auto px-4 sm:px-6 py-8">
        <div className="rounded-lg bg-red-50 border border-red-200 p-4 text-red-700">{error}</div>
      </div>
    );
  }

  if (!dashboard) return null;

  // Compute derived stats
  const approvedCount = dashboard.permitsByStatus.find(r => r.status === 'approved')?.count ?? 0;
  const pendingCount = dashboard.permitsByStatus
    .filter(r => ['submitted', 'under_review', 'additional_info_needed'].includes(r.status))
    .reduce((sum, r) => sum + r.count, 0);
  const totalReviewers = dashboard.reviewerWorkload.length;

  return (
    <div className="max-w-6xl mx-auto px-4 sm:px-6 py-8 space-y-6">
      <div>
        <h1 className="text-2xl font-bold text-slate-900">Admin Dashboard</h1>
        <p className="text-sm text-slate-500 mt-1">System-wide overview of all permit applications</p>
      </div>

      {/* 4 stat cards — 4 across on lg, 2 on sm, 1 on mobile */}
      <div className="grid grid-cols-1 sm:grid-cols-2 lg:grid-cols-4 gap-4">
        <DashboardStatCard
          label="Total Applications"
          value={dashboard.totalPermits}
          accent="blue"
        />
        <DashboardStatCard
          label="In Progress"
          value={pendingCount}
          accent={pendingCount > 0 ? 'amber' : 'slate'}
        />
        <DashboardStatCard
          label="Approved"
          value={approvedCount}
          accent="green"
        />
        <DashboardStatCard
          label="Active Reviewers"
          value={totalReviewers}
          accent="slate"
        />
      </div>

      {/* Chart + Workload table side by side on lg */}
      <div className="grid grid-cols-1 lg:grid-cols-2 gap-6">
        {/* Status distribution chart */}
        <div className="bg-white rounded-xl border border-slate-200 shadow-sm p-6">
          <h2 className="text-base font-semibold text-slate-900 mb-4">Applications by Status</h2>
          <StatusDistributionChart data={dashboard.permitsByStatus} title="All Applications" />
        </div>

        {/* Reviewer workload */}
        <div className="bg-white rounded-xl border border-slate-200 shadow-sm p-6">
          <h2 className="text-base font-semibold text-slate-900 mb-4">Reviewer Workload</h2>
          <ReviewerWorkloadTable rows={dashboard.reviewerWorkload} />
        </div>
      </div>

      {/* Recent system-wide activity feed */}
      <div className="bg-white rounded-xl border border-slate-200 shadow-sm p-6">
        <h2 className="text-base font-semibold text-slate-900 mb-4">Recent Activity</h2>
        {dashboard.recentActivity.length === 0 ? (
          <p className="text-sm text-slate-400 py-4 text-center">No recent activity</p>
        ) : (
          <ol className="divide-y divide-slate-100">
            {dashboard.recentActivity.map((item, idx) => (
              <li key={`${item.applicationId}-${idx}`} className="py-3 flex items-start gap-4">
                <div className="flex-1 min-w-0">
                  <p className="text-sm font-medium text-slate-800 truncate">{item.title}</p>
                  <p className="text-xs text-slate-500 mt-0.5">
                    <span className="font-medium text-slate-700">{item.changedByName}</span>
                    {' → '}
                    <span className="font-medium">{STATUS_LABELS[item.toStatus] ?? item.toStatus}</span>
                  </p>
                </div>
                <time className="text-xs text-slate-400 flex-shrink-0">{timeAgo(item.changedAt)}</time>
              </li>
            ))}
          </ol>
        )}
      </div>
    </div>
  );
}
```
