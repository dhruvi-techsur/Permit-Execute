```tsx
interface DashboardStatCardProps {
  label: string;
  value: number | string;
  icon?: React.ReactNode;       // optional SVG icon element
  accent?: 'blue' | 'amber' | 'green' | 'red' | 'slate';
  onClick?: () => void;         // optional click → navigate elsewhere
}

export function DashboardStatCard({ label, value, icon, accent = 'blue', onClick }: DashboardStatCardProps) {
  const ACCENT_STYLES = {
    blue:  'bg-blue-50 text-blue-600 border-blue-100',
    amber: 'bg-amber-50 text-amber-600 border-amber-100',
    green: 'bg-emerald-50 text-emerald-600 border-emerald-100',
    red:   'bg-red-50 text-red-600 border-red-100',
    slate: 'bg-slate-50 text-slate-600 border-slate-100',
  };

  return (
    <div
      className={`
        rounded-xl border p-5 flex flex-col gap-3 bg-white shadow-sm
        ${onClick ? 'cursor-pointer hover:shadow-md hover:-translate-y-0.5 transition-all duration-150' : ''}
      `}
      onClick={onClick}
      role={onClick ? 'button' : undefined}
    >
      {icon && (
        <div className={`w-10 h-10 rounded-lg flex items-center justify-center ${ACCENT_STYLES[accent]}`}>
          {icon}
        </div>
      )}
      <div>
        <p className="text-3xl font-bold text-slate-900">{value}</p>
        <p className="text-sm text-slate-500 mt-0.5">{label}</p>
      </div>
    </div>
  );
}
```

```tsx
import type { PermitStatusCount } from '../../types/dashboard.types';

const STATUS_COLORS: Record<string, string> = {
  draft:                  '#94a3b8', // slate-400
  submitted:              '#3b82f6', // blue-500
  under_review:           '#f59e0b', // amber-500
  additional_info_needed: '#f97316', // orange-500
  approved:               '#10b981', // emerald-500
  rejected:               '#ef4444', // red-500
};

const STATUS_LABELS: Record<string, string> = {
  draft:                  'Draft',
  submitted:              'Submitted',
  under_review:           'Under Review',
  additional_info_needed: 'Info Needed',
  approved:               'Approved',
  rejected:               'Rejected',
};

interface StatusDistributionChartProps {
  data: PermitStatusCount[];
  title?: string;
}

export function StatusDistributionChart({ data, title = 'Status Distribution' }: StatusDistributionChartProps) {
  if (!data || data.length === 0) {
    return (
      <div className="flex items-center justify-center h-32 text-slate-400 text-sm">No data yet</div>
    );
  }

  const total = data.reduce((sum, d) => sum + d.count, 0);
  if (total === 0) {
    return (
      <div className="flex items-center justify-center h-32 text-slate-400 text-sm">No permits yet</div>
    );
  }

  return (
    <div>
      <h3 className="text-sm font-medium text-slate-700 mb-3">{title}</h3>
      <div className="space-y-2">
        {data.map(({ status, count }) => {
          const pct = Math.round((count / total) * 100);
          return (
            <div key={status}>
              <div className="flex justify-between text-xs text-slate-600 mb-1">
                <span>{STATUS_LABELS[status] ?? status}</span>
                <span className="font-medium">{count} ({pct}%)</span>
              </div>
              <div className="h-2 rounded-full bg-slate-100 overflow-hidden">
                <div
                  className="h-full rounded-full transition-all duration-500"
                  style={{
                    width: `${pct}%`,
                    backgroundColor: STATUS_COLORS[status] ?? '#94a3b8',
                  }}
                />
              </div>
            </div>
          );
        })}
      </div>
      <p className="text-xs text-slate-400 mt-3 text-right">Total: {total}</p>
    </div>
  );
}
```

```tsx
import type { RecentStatusChange } from '../../types/dashboard.types';

const STATUS_LABELS: Record<string, string> = {
  draft:                  'Draft',
  submitted:              'Submitted',
  under_review:           'Under Review',
  additional_info_needed: 'Info Needed',
  approved:               'Approved',
  rejected:               'Rejected',
};

const STATUS_DOT_COLORS: Record<string, string> = {
  submitted:              'bg-blue-500',
  under_review:           'bg-amber-500',
  additional_info_needed: 'bg-orange-500',
  approved:               'bg-emerald-500',
  rejected:               'bg-red-500',
  draft:                  'bg-slate-400',
};

function timeAgo(dateStr: string): string {
  const diffMs = Date.now() - new Date(dateStr).getTime();
  const diffMins = Math.floor(diffMs / 60_000);
  if (diffMins < 1) return 'just now';
  if (diffMins < 60) return `${diffMins}m ago`;
  const diffHrs = Math.floor(diffMins / 60);
  if (diffHrs < 24) return `${diffHrs}h ago`;
  const diffDays = Math.floor(diffHrs / 24);
  return `${diffDays}d ago`;
}

interface RecentActivityFeedProps {
  items: RecentStatusChange[];
  emptyText?: string;
}

export function RecentActivityFeed({ items, emptyText = 'No recent activity' }: RecentActivityFeedProps) {
  if (items.length === 0) {
    return <p className="text-sm text-slate-400 py-4 text-center">{emptyText}</p>;
  }

  return (
    <ol className="space-y-3">
      {items.map((item, idx) => (
        <li key={`${item.applicationId}-${idx}`} className="flex items-start gap-3">
          <span className={`mt-1.5 w-2 h-2 rounded-full flex-shrink-0 ${STATUS_DOT_COLORS[item.toStatus] ?? 'bg-slate-400'}`} />
          <div className="flex-1 min-w-0">
            <p className="text-sm text-slate-800 truncate font-medium">{item.title}</p>
            <p className="text-xs text-slate-500">
              → <span className="font-medium">{STATUS_LABELS[item.toStatus] ?? item.toStatus}</span>
              {' · '}
              <time>{timeAgo(item.changedAt)}</time>
            </p>
          </div>
        </li>
      ))}
    </ol>
  );
}
```
