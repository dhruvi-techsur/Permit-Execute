**`frontend/src/store/dashboard.store.ts`** — Zustand store:
```typescript
import { create } from 'zustand';
import type { ApplicantDashboardDto, ReviewerDashboardDto, AdminDashboardDto } from '../types/dashboard.types';
import { getApplicantDashboard, getReviewerDashboard, getAdminDashboard } from '../api/dashboard.api';

interface DashboardState {
  applicantData: ApplicantDashboardDto | null;
  reviewerData: ReviewerDashboardDto | null;
  adminData: AdminDashboardDto | null;
  isLoading: boolean;
  error: string | null;
  fetchApplicantDashboard: () => Promise<void>;
  fetchReviewerDashboard: () => Promise<void>;
  fetchAdminDashboard: () => Promise<void>;
  clearDashboard: () => void;
}

export const useDashboardStore = create<DashboardState>((set) => ({
  applicantData: null,
  reviewerData: null,
  adminData: null,
  isLoading: false,
  error: null,

  fetchApplicantDashboard: async () => {
    set({ isLoading: true, error: null });
    try {
      const data = await getApplicantDashboard();
      set({ applicantData: data, isLoading: false });
    } catch (err: unknown) {
      set({ isLoading: false, error: (err as Error).message });
    }
  },

  fetchReviewerDashboard: async () => {
    set({ isLoading: true, error: null });
    try {
      const data = await getReviewerDashboard();
      set({ reviewerData: data, isLoading: false });
    } catch (err: unknown) {
      set({ isLoading: false, error: (err as Error).message });
    }
  },

  fetchAdminDashboard: async () => {
    set({ isLoading: true, error: null });
    try {
      const data = await getAdminDashboard();
      set({ adminData: data, isLoading: false });
    } catch (err: unknown) {
      set({ isLoading: false, error: (err as Error).message });
    }
  },

  clearDashboard: () => set({ applicantData: null, reviewerData: null, adminData: null, error: null }),
}));
```

**`frontend/src/hooks/useDashboard.ts`** — role-aware fetch hook:
```typescript
import { useEffect, useCallback } from 'react';
import { useAuthStore } from '../store/auth.store';
import { useDashboardStore } from '../store/dashboard.store';
import type { DashboardData } from '../types/dashboard.types';

export function useDashboard(): {
  data: DashboardData | null;
  isLoading: boolean;
  error: string | null;
  refresh: () => void;
} {
  const { user } = useAuthStore();
  const {
    applicantData,
    reviewerData,
    adminData,
    isLoading,
    error,
    fetchApplicantDashboard,
    fetchReviewerDashboard,
    fetchAdminDashboard,
  } = useDashboardStore();

  const fetchForRole = useCallback(() => {
    if (!user) return;
    if (user.role === 'applicant') fetchApplicantDashboard();
    else if (user.role === 'reviewer') fetchReviewerDashboard();
    else if (user.role === 'admin') fetchAdminDashboard();
  }, [user, fetchApplicantDashboard, fetchReviewerDashboard, fetchAdminDashboard]);

  // Fetch on mount and when role changes
  useEffect(() => {
    fetchForRole();
  }, [fetchForRole]);

  // Derive the correct data slice for the current user's role
  const data: DashboardData | null =
    user?.role === 'applicant' ? applicantData :
    user?.role === 'reviewer' ? reviewerData :
    user?.role === 'admin' ? adminData :
    null;

  return { data, isLoading, error, refresh: fetchForRole };
}
```
