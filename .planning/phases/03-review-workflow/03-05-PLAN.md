```tsx
const STATUS_STYLES: Record<ReviewStatus | 'draft', string> = {
  draft: 'bg-slate-100 text-slate-600',
  submitted: 'bg-blue-100 text-blue-700',
  under_review: 'bg-amber-100 text-amber-700',
  additional_info_needed: 'bg-orange-100 text-orange-700',
  approved: 'bg-emerald-100 text-emerald-700',
  rejected: 'bg-red-100 text-red-700',

// Readable labels
const STATUS_LABELS: Record<string, string> = {
  submitted: 'Submitted',
  under_review: 'Under Review',
  additional_info_needed: 'Info Needed',
  approved: 'Approved',
  rejected: 'Rejected',
};

export function ReviewStatusBadge({ status }: { status: string }) {
  return (
    <span className={`inline-flex items-center px-2.5 py-0.5 rounded-full text-xs font-medium ${STATUS_STYLES[status] || 'bg-slate-100 text-slate-600'}`}>
      {STATUS_LABELS[status] || status}
    </span>
  );
}
```

```tsx
export default function ReviewListPage() {
  const { applications, isLoadingList, listError, fetchApplications } = useReviewStore();
  const { user } = useAuthStore();
  const navigate = useNavigate();
  const [filters, setFilters] = useState<ReviewerQueryParams>({ page: 1, limit: 20 });

  useEffect(() => { fetchApplications(filters); }, [filters]);

  return (
    <div className="max-w-6xl mx-auto px-4 sm:px-6 py-8">
      <div className="flex items-center justify-between mb-6">
        <h1 className="text-2xl font-bold text-slate-900">Applications for Review</h1>
        {/* Filter controls */}
        <div className="flex gap-3">
          <select onChange={e => setFilters(f => ({ ...f, status: e.target.value as ReviewStatus || undefined, page: 1 }))}>
            <option value="">All Statuses</option>
            <option value="submitted">Submitted</option>
            <option value="under_review">Under Review</option>
            <option value="additional_info_needed">Info Needed</option>
            <option value="approved">Approved</option>
            <option value="rejected">Rejected</option>
          </select>
          <label className="flex items-center gap-2 text-sm">
            <input type="checkbox" onChange={e => setFilters(f => ({ ...f, assignedToMe: e.target.checked, page: 1 }))} />
            Assigned to me
          </label>
        </div>
      </div>

      {/* Skeleton loading — 3 skeleton cards */}
      {isLoadingList && (
        <div className="space-y-4">
          {[1,2,3].map(i => <div key={i} className="h-28 rounded-xl bg-slate-100 animate-pulse" />)}
        </div>
      )}

      {/* Error state */}
      {listError && !isLoadingList && (
        <div className="rounded-lg bg-red-50 border border-red-200 p-4 text-red-700">{listError}</div>
      )}

      {/* Empty state */}
      {!isLoadingList && !listError && applications.length === 0 && (
        <div className="text-center py-16 text-slate-500">
          <p className="text-lg font-medium">No applications found</p>
          <p className="text-sm">Try adjusting your filters</p>
        </div>
      )}

      {/* Application list */}
      {!isLoadingList && !listError && applications.length > 0 && (
        <div className="space-y-3">
          {applications.map(app => (
            <ReviewApplicationCard
              key={app.id}
              application={app}
              currentUserId={user?.id}
              onClick={() => navigate(`/review/applications/${app.id}`)}
            />
          ))}
        </div>
      )}
    </div>
  );
}
```

**ReviewDetailPage** (`/review/applications/:id`):
```tsx
export default function ReviewDetailPage() {
  const { id } = useParams<{ id: string }>();
  const { selectedDetail, isLoadingDetail, detailError, fetchDetail, clearDetail } = useReviewStore();
  const navigate = useNavigate();

  useEffect(() => {
    if (id) fetchDetail(id);
    return () => clearDetail();
  }, [id]);

  // Skeleton: 2 skeleton blocks during load
  if (isLoadingDetail) {
    return (
      <div className="max-w-5xl mx-auto px-4 sm:px-6 py-8 space-y-6">
        <div className="h-8 bg-slate-100 animate-pulse rounded-lg w-64" />
        <div className="h-48 bg-slate-100 animate-pulse rounded-xl" />
        <div className="h-64 bg-slate-100 animate-pulse rounded-xl" />
      </div>
    );
  }

  if (detailError) {
    return <div className="text-center py-16 text-red-600">{detailError}</div>;
  }

  if (!selectedDetail) return null;

  const { application, documents, history } = selectedDetail;

  return (
    <div className="max-w-5xl mx-auto px-4 sm:px-6 py-8">
      {/* Back button */}
      <button onClick={() => navigate('/review/applications')} className="flex items-center gap-1 text-sm text-slate-500 hover:text-slate-900 mb-6 transition-colors">
        ← Back to Applications
      </button>

      {/* Header */}
      <div className="flex items-start justify-between mb-6">
        <div>
          <h1 className="text-2xl font-bold text-slate-900">{application.title}</h1>
          <p className="text-sm text-slate-500 mt-1">
            {application.permitType} · {application.siteAddress}
          </p>
        </div>
        <ReviewStatusBadge status={application.status} />
      </div>

      {/* Two-column layout: left = details + docs + history, right = actions panel + messaging slot */}
      <div className="grid grid-cols-1 lg:grid-cols-3 gap-6">
        {/* Left column (2/3) */}
        <div className="lg:col-span-2 space-y-6">
          {/* Application metadata card */}
          <section className="bg-white rounded-xl border border-slate-200 shadow-sm p-6">
            <h2 className="text-lg font-semibold text-slate-900 mb-4">Application Details</h2>
            <dl className="grid grid-cols-2 gap-4 text-sm">
              <div><dt className="text-slate-500">Applicant ID</dt><dd className="font-medium">{application.applicantId}</dd></div>
              <div><dt className="text-slate-500">Submitted</dt><dd className="font-medium">{application.submissionDate ? new Date(application.submissionDate).toLocaleDateString() : 'Not submitted'}</dd></div>
              <div className="col-span-2"><dt className="text-slate-500">Description</dt><dd className="font-medium">{application.description}</dd></div>
              {application.decisionReason && (
                <div className="col-span-2"><dt className="text-slate-500">Decision Reason</dt><dd className="font-medium">{application.decisionReason}</dd></div>
              )}
            </dl>
          </section>

          {/* Status timeline */}
          <section className="bg-white rounded-xl border border-slate-200 shadow-sm p-6">
            <h2 className="text-lg font-semibold text-slate-900 mb-4">Status History</h2>
            {history.length === 0 ? (
              <p className="text-slate-500 text-sm">No history yet</p>
            ) : (
              <ol className="relative border-l border-slate-200 space-y-4">
                {history.map(entry => (
                  <li key={entry.id} className="ml-4">
                    <div className="absolute -left-1.5 w-3 h-3 rounded-full bg-primary-500 border-2 border-white" />
                    <ReviewStatusBadge status={entry.toStatus} />
                    {entry.note && <p className="text-sm text-slate-600 mt-1">{entry.note}</p>}
                    <time className="text-xs text-slate-400">{new Date(entry.createdAt).toLocaleString()}</time>
                  </li>
                ))}
              </ol>
            )}
          </section>

          {/* Documents */}
          <section className="bg-white rounded-xl border border-slate-200 shadow-sm p-6">
            <h2 className="text-lg font-semibold text-slate-900 mb-4">Documents ({documents.length})</h2>
            <DocumentDownloadList documents={documents} />
          </section>
        </div>

        {/* Right column (1/3) — actions + messaging slot */}
        <div className="space-y-6">
          <section className="bg-white rounded-xl border border-slate-200 shadow-sm p-6">
            <h2 className="text-lg font-semibold text-slate-900 mb-4">Review Actions</h2>
            <StatusActionPanel application={application} applicationId={application.id} />
          </section>

          {/* Messaging panel slot — filled by plan 03-06 */}
          <div id="messaging-panel-slot" />
        </div>
      </div>
    </div>
  );
}
```

| T-03-24 | Denial of service | Rapid action button clicks (double-submit) | mitigate | `isPerformingAction` flag in useReviewStore disables action buttons while an API call is in-flight — prevents duplicate status transition requests |
