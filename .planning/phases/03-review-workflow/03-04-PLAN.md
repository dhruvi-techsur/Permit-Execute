```typescript
import { create } from 'zustand';
import { PermitApplication } from '../types/permit.types';
import { ReviewApplicationDetail, ReviewerQueryParams, PaginatedReviewApplications, RequestInfoPayload, DecisionPayload } from '../types/review.types';
import * as reviewApi from '../api/review.api';

interface ReviewState {
  // List state
  applications: PermitApplication[];
  totalApplications: number;
  currentPage: number;
  isLoadingList: boolean;
  listError: string | null;

  // Detail state
  selectedDetail: ReviewApplicationDetail | null;
  isLoadingDetail: boolean;
  detailError: string | null;

  // Action state (for status transitions)
  isPerformingAction: boolean;
  actionError: string | null;

  // Actions
  fetchApplications: (params?: ReviewerQueryParams) => Promise<void>;
  fetchDetail: (id: string) => Promise<void>;
  assignReviewer: (id: string) => Promise<PermitApplication>;
  advanceToReview: (id: string) => Promise<PermitApplication>;
  requestInfo: (id: string, payload: RequestInfoPayload) => Promise<PermitApplication>;
  respondToInfoRequest: (id: string) => Promise<PermitApplication>;
  approveApplication: (id: string, payload: DecisionPayload) => Promise<PermitApplication>;
  rejectApplication: (id: string, payload: DecisionPayload) => Promise<PermitApplication>;
  clearDetail: () => void;
  clearErrors: () => void;
}

export const useReviewStore = create<ReviewState>((set, get) => ({
  applications: [],
  totalApplications: 0,
  currentPage: 1,
  isLoadingList: false,
  listError: null,
  selectedDetail: null,
  isLoadingDetail: false,
  detailError: null,
  isPerformingAction: false,
  actionError: null,

  fetchApplications: async (params = {}) => {
    set({ isLoadingList: true, listError: null });
    try {
      const result = await reviewApi.listReviewApplications(params);
      set({ applications: result.data, totalApplications: result.meta.total, currentPage: result.meta.page, isLoadingList: false });
    } catch (err: any) {
      set({ isLoadingList: false, listError: err?.response?.data?.message || 'Failed to load applications' });
    }
  },

  fetchDetail: async (id) => {
    set({ isLoadingDetail: true, detailError: null });
    try {
      const detail = await reviewApi.getReviewApplicationDetail(id);
      set({ selectedDetail: detail, isLoadingDetail: false });
    } catch (err: any) {
      set({ isLoadingDetail: false, detailError: err?.response?.data?.message || 'Failed to load detail' });
    }
  },

  // Generic action wrapper used by all status transitions
  assignReviewer: async (id) => {
    set({ isPerformingAction: true, actionError: null });
    try {
      const updated = await reviewApi.assignReviewer(id);
      // Update the application in the list if present
      set(state => ({
        applications: state.applications.map(a => a.id === id ? updated : a),
        isPerformingAction: false,
      }));
      return updated;
    } catch (err: any) {
      set({ isPerformingAction: false, actionError: err?.response?.data?.message || 'Action failed' });
      throw err;
    }
  },

  advanceToReview: async (id) => {
    set({ isPerformingAction: true, actionError: null });
    try {
      const updated = await reviewApi.advanceToReview(id);
      set(state => ({
        applications: state.applications.map(a => a.id === id ? updated : a),
        selectedDetail: state.selectedDetail ? { ...state.selectedDetail, application: updated } : null,
        isPerformingAction: false,
      }));
      return updated;
    } catch (err: any) {
      set({ isPerformingAction: false, actionError: err?.response?.data?.message || 'Action failed' });
      throw err;
    }
  },

  requestInfo: async (id, payload) => {
    set({ isPerformingAction: true, actionError: null });
    try {
      const updated = await reviewApi.requestInfo(id, payload);
      set(state => ({
        applications: state.applications.map(a => a.id === id ? updated : a),
        selectedDetail: state.selectedDetail ? { ...state.selectedDetail, application: updated } : null,
        isPerformingAction: false,
      }));
      return updated;
    } catch (err: any) {
      set({ isPerformingAction: false, actionError: err?.response?.data?.message || 'Action failed' });
      throw err;
    }
  },

  respondToInfoRequest: async (id) => {
    set({ isPerformingAction: true, actionError: null });
    try {
      const updated = await reviewApi.respondToInfoRequest(id);
      set({ isPerformingAction: false });
      return updated;
    } catch (err: any) {
      set({ isPerformingAction: false, actionError: err?.response?.data?.message || 'Action failed' });
      throw err;
    }
  },

  approveApplication: async (id, payload) => {
    set({ isPerformingAction: true, actionError: null });
    try {
      const updated = await reviewApi.approveApplication(id, payload);
      set(state => ({
        applications: state.applications.map(a => a.id === id ? updated : a),
        selectedDetail: state.selectedDetail ? { ...state.selectedDetail, application: updated } : null,
        isPerformingAction: false,
      }));
      return updated;
    } catch (err: any) {
      set({ isPerformingAction: false, actionError: err?.response?.data?.message || 'Action failed' });
```

```typescript
import { create } from 'zustand';
import { Message, SendMessagePayload } from '../types/messaging.types';
import * as messagingApi from '../api/messaging.api';

interface MessagingState {
  // Keyed by applicationId
  threads: Record<string, Message[]>;
  unreadCounts: Record<string, number>;
  isLoadingThread: Record<string, boolean>;
  isSending: boolean;
  sendError: string | null;

  fetchThread: (permitId: string) => Promise<void>;
  sendMessage: (permitId: string, payload: SendMessagePayload) => Promise<Message>;
  fetchUnreadCount: (permitId: string) => Promise<void>;
  markAllRead: (permitId: string) => Promise<void>;
  clearThread: (permitId: string) => void;
}

export const useMessagingStore = create<MessagingState>((set, get) => ({
  threads: {},
  unreadCounts: {},
  isLoadingThread: {},
  isSending: false,
  sendError: null,

  fetchThread: async (permitId) => {
    set(state => ({ isLoadingThread: { ...state.isLoadingThread, [permitId]: true } }));
    try {
      const thread = await messagingApi.listMessages(permitId);
      set(state => ({
        threads: { ...state.threads, [permitId]: thread.data },
        isLoadingThread: { ...state.isLoadingThread, [permitId]: false },
      }));
    } catch {
      set(state => ({ isLoadingThread: { ...state.isLoadingThread, [permitId]: false } }));
    }
  },

  sendMessage: async (permitId, payload) => {
    set({ isSending: true, sendError: null });
    try {
      const message = await messagingApi.sendMessage(permitId, payload);
      // Optimistically append — polling will reconcile if needed
      set(state => ({
        threads: { ...state.threads, [permitId]: [...(state.threads[permitId] || []), message] },
        isSending: false,
      }));
      return message;
    } catch (err: any) {
      set({ isSending: false, sendError: err?.response?.data?.message || 'Failed to send message' });
      throw err;
    }
  },

  fetchUnreadCount: async (permitId) => {
    try {
      const { count } = await messagingApi.getUnreadCount(permitId);
      set(state => ({ unreadCounts: { ...state.unreadCounts, [permitId]: count } }));
    } catch {
      // Silently fail — unread count is cosmetic
    }
  },

  markAllRead: async (permitId) => {
    await messagingApi.markAllRead(permitId);
    set(state => ({ unreadCounts: { ...state.unreadCounts, [permitId]: 0 } }));
  },

  clearThread: (permitId) => set(state => {
    const { [permitId]: _, ...threads } = state.threads;
    const { [permitId]: __, ...counts } = state.unreadCounts;
    return { threads, unreadCounts: counts };
  }),
}));
```

```typescript
import { useEffect, useRef } from 'react';
import { useMessagingStore } from '../store/messaging.store';

const POLL_INTERVAL_MS = 10_000;

export function useMessageThread(permitId: string) {
  const { threads, isLoadingThread, isSending, sendError, fetchThread, sendMessage, fetchUnreadCount, markAllRead } = useMessagingStore();
  const intervalRef = useRef<ReturnType<typeof setInterval> | null>(null);

  useEffect(() => {
    if (!permitId) return;
    // Initial fetch
    fetchThread(permitId);
    fetchUnreadCount(permitId);

    // Poll every 10 seconds
    intervalRef.current = setInterval(() => {
      fetchThread(permitId);
      fetchUnreadCount(permitId);
    }, POLL_INTERVAL_MS);

    return () => {
      if (intervalRef.current) clearInterval(intervalRef.current);
    };
  }, [permitId]);

  return {
    messages: threads[permitId] || [],
    isLoading: isLoadingThread[permitId] || false,
    isSending,
    sendError,
    sendMessage: (body: string) => sendMessage(permitId, { body }),
    markAllRead: () => markAllRead(permitId),
  };
}
```
