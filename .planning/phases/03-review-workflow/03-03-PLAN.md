```typescript
export enum MessageType {
  APPLICANT_MESSAGE = 'applicant_message',
  REVIEWER_MESSAGE = 'reviewer_message',
  STATUS_NOTE = 'status_note',
  INFO_REQUEST = 'info_request',
}

@Entity('messages')
export class Message {
  @PrimaryGeneratedColumn('uuid') id: string;
  @Column({ name: 'application_id' }) applicationId: string;
  @Column({ name: 'sender_id' }) senderId: string;
  @Column() body: string;
  @Column({ type: 'enum', enum: MessageType, name: 'message_type' }) messageType: MessageType;
  @Column({ name: 'has_attachment', default: false }) hasAttachment: boolean;
  @CreateDateColumn({ name: 'created_at' }) createdAt: Date;
  @UpdateDateColumn({ name: 'updated_at' }) updatedAt: Date;
  // Joined at query time (not a FK relation in entity to avoid N+1)
  senderName?: string;
  senderRole?: string;
}
```

**MessagingService** — inject: `@InjectRepository(Message)`, `@InjectRepository(MessageAttachment)`, `@InjectRepository(PermitApplication)`, `@InjectRepository(User)`, `S3Service`, `NotificationsService`.

Methods:

`verifyApplicationAccess(applicationId: string, userId: string, userRole: string)`:
- Find permit by applicationId, throw NotFoundException if not found
- If role === 'applicant': throw ForbiddenException if permit.applicantId !== userId
- If role === 'reviewer' or 'admin': throw ForbiddenException if status === 'draft' (cannot message on draft)
- Return permit

`sendMessage(applicationId: string, senderId: string, senderRole: string, dto: SendMessageDto)`:
- Call verifyApplicationAccess
- Determine messageType from role: 'applicant' → MessageType.APPLICANT_MESSAGE, 'reviewer' → MessageType.REVIEWER_MESSAGE
- Insert message: { applicationId, senderId, body: dto.body, messageType, hasAttachment: false }
- Determine recipient: if sender is applicant → notify permit.reviewerId (if set); if sender is reviewer → notify permit.applicantId
- Create notification: type='message_received', title='New Message', body=`You have a new message on application "${permit.title}".`, applicationId
- Return message with sender name and role resolved via user join

`listMessages(applicationId: string, userId: string, userRole: string, query: MessageQueryDto)`:
- Call verifyApplicationAccess
- Query messages WHERE application_id = applicationId
- JOIN users ON users.id = messages.sender_id to get full_name and role
- Order by created_at ASC (oldest first — chat thread order)
- Paginate
- Map result to include senderName and senderRole on each message
- Return `{ data: MessageWithSender[], meta: { total, page, limit } }`

`presignAttachment(applicationId: string, userId: string, userRole: string, dto: MessagePresignDto)`:
- Call verifyApplicationAccess
- Find message by dto.messageId, verify message.senderId === userId (own message only)
- Generate storageKey: `permit-messages/${applicationId}/${message.id}/${uuidv4()}/${dto.fileName}`
- Insert MessageAttachment row: { messageId: dto.messageId, fileName: dto.fileName, fileType: dto.mimeType, fileSizeBytes: dto.fileSizeBytes, storageKey }
- Call S3Service.generateUploadUrl(storageKey, dto.mimeType, 900)
- Return `{ attachmentId: attachment.id, uploadUrl, storageKey }`

`confirmAttachment(messageId: string, attachmentId: string, userId: string)`:
- Find message by messageId, verify senderId === userId
- Find attachment by attachmentId + messageId
- Update message.hasAttachment = true
- Return attachment

`getUnreadCount(applicationId: string, userId: string, userRole: string)`:
- Call verifyApplicationAccess
- Count messages WHERE application_id = applicationId AND sender_id != userId
  (messages from the OTHER party that the current user hasn't "read")
  Note: We track unread at the thread level — all messages from the other party are "unread" until mark-read is called. For this implementation, unread = messages from non-self senders where created_at > last_read_at. Since we don't have a per-user read cursor, use a simple approximation: count messages from other senders in the last 24 hours (or all unread if there's no read cursor).
  **Simpler approach:** count messages WHERE application_id = applicationId AND sender_id != userId — this overestimates but mark-read resets the concept. Return `{ count }`.

`markAllRead(applicationId: string, userId: string, userRole: string)`:
- Call verifyApplicationAccess
- (This endpoint semantically marks the thread as read for the user — since we don't have a per-user read cursor table in v1, this is a no-op that returns 204 successfully. The client can use this to reset its local unread badge.)
- Return void

**MessagingController** (`messaging.controller.ts`):

```typescript
@Controller('api/v1/permits/:permitId/messages')
@UseGuards(JwtAuthGuard)
export class MessagingController {
  constructor(private readonly messagingService: MessagingService) {}

  // POST /api/v1/permits/:permitId/messages — both applicant and reviewer
  @Post()
  @HttpCode(201)
  sendMessage(
    @Param('permitId', ParseUUIDPipe) permitId: string,
    @Body() dto: SendMessageDto,
    @Request() req,
  ) {
    return this.messagingService.sendMessage(permitId, req.user.id, req.user.role, dto);
  }

  // GET /api/v1/permits/:permitId/messages
  @Get()
  listMessages(
    @Param('permitId', ParseUUIDPipe) permitId: string,
    @Query() query: MessageQueryDto,
    @Request() req,
  ) {
    return this.messagingService.listMessages(permitId, req.user.id, req.user.role, query);
  }

  // GET /api/v1/permits/:permitId/messages/unread-count
  // IMPORTANT: declare before :msgId routes to avoid NestJS route collision
  @Get('unread-count')
  getUnreadCount(
    @Param('permitId', ParseUUIDPipe) permitId: string,
    @Request() req,
  ) {
    return this.messagingService.getUnreadCount(permitId, req.user.id, req.user.role);
  }

  // PATCH /api/v1/permits/:permitId/messages/mark-read
  @Patch('mark-read')
  @HttpCode(204)
  markAllRead(
    @Param('permitId', ParseUUIDPipe) permitId: string,
    @Request() req,
  ) {
    return this.messagingService.markAllRead(permitId, req.user.id, req.user.role);
  }

  // POST /api/v1/permits/:permitId/messages/presign — reviewer only (MSG-04)
  @Post('presign')
  @Roles('reviewer', 'admin')
  @UseGuards(RolesGuard)
  @HttpCode(201)
  presignAttachment(
    @Param('permitId', ParseUUIDPipe) permitId: string,
    @Body() dto: MessagePresignDto,
    @Request() req,
  ) {
    return this.messagingService.presignAttachment(permitId, req.user.id, req.user.role, dto);
  }

  // POST /api/v1/permits/:permitId/messages/:msgId/confirm-attachment
  @Post(':msgId/confirm-attachment')
  confirmAttachment(
    @Param('permitId', ParseUUIDPipe) permitId: string,
    @Param('msgId', ParseUUIDPipe) msgId: string,
    @Body('attachmentId', ParseUUIDPipe) attachmentId: string,
    @Request() req,
  ) {
    return this.messagingService.confirmAttachment(msgId, attachmentId, req.user.id);
  }
}
```

**IMPORTANT route ordering:** The controller method order matters in NestJS routing. Static paths (`unread-count`, `mark-read`, `presign`) MUST be declared before parametric paths (`:msgId`) in the class body.

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| client→API | Message body and attachment metadata entering messaging handlers |
| API→S3 | Presigned URL for attachment upload — client PUTs directly to MinIO |
| DB→response | Message thread (potentially confidential permit discussion) returned to authenticated party |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-03-11 | Elevation of privilege | GET/POST /permits/:id/messages | mitigate | `messaging.service.ts::verifyApplicationAccess` enforces: applicant can only message their own application; reviewer cannot message on draft applications. Checked before every operation. |
| T-03-12 | Elevation of privilege | POST /permits/:id/messages messageType | mitigate | `messageType` is derived server-side from `req.user.role` in `MessagingService.sendMessage()` — client cannot supply or override it; prevents applicant from sending a message with type 'reviewer_message' |
| T-03-13 | Elevation of privilege | POST /permits/:id/messages/presign | mitigate | `@Roles('reviewer', 'admin') @UseGuards(RolesGuard)` restricts attachment presign to reviewers only (MSG-04 requirement); applicants cannot create message attachments |
| T-03-14 | Elevation of privilege | POST /permits/:id/messages/:msgId/confirm-attachment | mitigate | `messaging.service.ts::confirmAttachment` verifies `message.senderId === userId` — only the message author can confirm attachments to their own message |
| T-03-15 | Tampering | message body length | mitigate | `SendMessageDto` uses `@MaxLength(10000)` — prevents excessively large messages from bloating the messages table |
| T-03-16 | Tampering | attachment storageKey path | mitigate | storageKey is server-generated as `permit-messages/${applicationId}/${message.id}/${uuidv4()}/${fileName}` — client never controls the path; prevents path traversal |
| T-03-17 | Information disclosure | GET /permits/:id/messages | mitigate | `verifyApplicationAccess` gates all list reads — only the permit's applicant or any reviewer/admin can read the thread; third parties cannot enumerate messages |
</threat_model>
