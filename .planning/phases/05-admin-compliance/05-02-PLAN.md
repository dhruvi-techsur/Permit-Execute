**`frontend/src/api/admin.api.ts`** — Axios functions using `apiClient` from `../lib/axios`:

```typescript
import apiClient from '../lib/axios';
import type {
  AdminPaginatedResponse, AdminUserDto, PermitApplicationSummary, AuditLogEntry,
  ApplicationsQueryParams, UsersQueryParams, AuditLogQueryParams,
  CreateUserPayload, UpdateUserPayload,
} from '../types/admin.types';

export const adminApi = {
  getApplications: (params?: ApplicationsQueryParams) =>
    apiClient.get<AdminPaginatedResponse<PermitApplicationSummary>>('/api/v1/admin/applications', { params })
      .then(r => r.data),

  getUsers: (params?: UsersQueryParams) =>
    apiClient.get<AdminPaginatedResponse<AdminUserDto>>('/api/v1/admin/users', { params })
      .then(r => r.data),

  createUser: (payload: CreateUserPayload) =>
    apiClient.post<AdminUserDto>('/api/v1/admin/users', payload)
      .then(r => r.data),

  updateUser: (id: string, payload: UpdateUserPayload) =>
    apiClient.patch<AdminUserDto>(`/api/v1/admin/users/${id}`, payload)
      .then(r => r.data),

  deactivateUser: (id: string) =>
    apiClient.delete<{ message: string }>(`/api/v1/admin/users/${id}`)
      .then(r => r.data),

  assignReviewer: (applicationId: string, reviewerId: string) =>
    apiClient.patch<{ applicationId: string; reviewerId: string; reviewerName: string }>(
      `/api/v1/admin/applications/${applicationId}/assign-reviewer`,
      { reviewerId },
    ).then(r => r.data),

  getAuditLog: (params?: AuditLogQueryParams) =>
    apiClient.get<AdminPaginatedResponse<AuditLogEntry>>('/api/v1/admin/audit-log', { params })
      .then(r => r.data),
};
```

**`frontend/src/store/admin.store.ts`** — Zustand store:

```typescript
import { create } from 'zustand';
import { adminApi } from '../api/admin.api';
import type { AdminUserDto, PermitApplicationSummary, AuditLogEntry, PaginationMeta, ApplicationsQueryParams, UsersQueryParams, AuditLogQueryParams } from '../types/admin.types';

interface AdminStore {
  applications: PermitApplicationSummary[];
  applicationsMeta: PaginationMeta | null;
  users: AdminUserDto[];
  usersMeta: PaginationMeta | null;
  auditLog: AuditLogEntry[];
  auditLogMeta: PaginationMeta | null;
  loading: boolean;
  error: string | null;
  fetchApplications: (params?: ApplicationsQueryParams) => Promise<void>;
  fetchUsers: (params?: UsersQueryParams) => Promise<void>;
  fetchAuditLog: (params?: AuditLogQueryParams) => Promise<void>;
  assignReviewer: (applicationId: string, reviewerId: string) => Promise<void>;
}

export const useAdminStore = create<AdminStore>((set) => ({
  applications: [], applicationsMeta: null,
  users: [], usersMeta: null,
  auditLog: [], auditLogMeta: null,
  loading: false, error: null,

  fetchApplications: async (params) => {
    set({ loading: true, error: null });
    try {
      const result = await adminApi.getApplications(params);
      set({ applications: result.data, applicationsMeta: result.meta });
    } catch (e: any) {
      set({ error: e?.response?.data?.message ?? 'Failed to load applications' });
    } finally { set({ loading: false }); }
  },

  fetchUsers: async (params) => {
    set({ loading: true, error: null });
    try {
      const result = await adminApi.getUsers(params);
      set({ users: result.data, usersMeta: result.meta });
    } catch (e: any) {
      set({ error: e?.response?.data?.message ?? 'Failed to load users' });
    } finally { set({ loading: false }); }
  },

  fetchAuditLog: async (params) => {
    set({ loading: true, error: null });
    try {
      const result = await adminApi.getAuditLog(params);
      set({ auditLog: result.data, auditLogMeta: result.meta });
    } catch (e: any) {
      set({ error: e?.response?.data?.message ?? 'Failed to load audit log' });
    } finally { set({ loading: false }); }
  },

  assignReviewer: async (applicationId, reviewerId) => {
    const result = await adminApi.assignReviewer(applicationId, reviewerId);
    set((state) => ({
      applications: state.applications.map((app) =>
        app.id === applicationId
          ? { ...app, reviewerName: result.reviewerName }
          : app,
      ),
    }));
  },
}));
```
